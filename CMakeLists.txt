cmake_minimum_required(VERSION 3.16)
project(sdl3_test_chess_app C)

include(CTest)

set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

find_package(Threads REQUIRED)
find_package(PkgConfig)

find_package(SDL3 REQUIRED CONFIG)
find_package(SDL3_ttf REQUIRED CONFIG)
find_package(SDL3_image REQUIRED CONFIG)

if(PkgConfig_FOUND)
    pkg_check_modules(AVAHI_CLIENT QUIET avahi-client)
endif()

# ── Embed PNG piece sprites into the binary ──────────────────────
set(PIECE_PNGS
    Chess_bdt60 Chess_blt60
    Chess_kdt60 Chess_klt60
    Chess_ndt60 Chess_nlt60
    Chess_pdt60 Chess_plt60
    Chess_qdt60 Chess_qlt60
    Chess_rdt60 Chess_rlt60
)

set(GENERATED_DIR "${CMAKE_CURRENT_BINARY_DIR}/generated")
file(MAKE_DIRECTORY "${GENERATED_DIR}")

set(EMBEDDED_SOURCES)
set(EMBEDDED_HEADER_LINES "/* Generated by CMake — do not edit. */\n#ifndef EMBEDDED_PIECES_H\n#define EMBEDDED_PIECES_H\n\n#include <stddef.h>\n")

foreach(piece ${PIECE_PNGS})
    set(_input  "${CMAKE_CURRENT_SOURCE_DIR}/assets/pieces/${piece}.png")
    set(_output "${GENERATED_DIR}/${piece}.c")
    set(_var    "embedded_${piece}")

    add_custom_command(
        OUTPUT  "${_output}"
        COMMAND "${CMAKE_COMMAND}"
                -DINPUT_FILE=${_input}
                -DOUTPUT_FILE=${_output}
                -DVAR_NAME=${_var}
                -P "${CMAKE_CURRENT_SOURCE_DIR}/cmake/bin2c.cmake"
        DEPENDS "${_input}" "${CMAKE_CURRENT_SOURCE_DIR}/cmake/bin2c.cmake"
        COMMENT "bin2c: ${piece}.png"
    )
    list(APPEND EMBEDDED_SOURCES "${_output}")
    string(APPEND EMBEDDED_HEADER_LINES "extern const unsigned char ${_var}[];\nextern const size_t ${_var}_size;\n")
endforeach()

string(APPEND EMBEDDED_HEADER_LINES "\n#endif /* EMBEDDED_PIECES_H */\n")
file(WRITE "${GENERATED_DIR}/embedded_pieces.h" "${EMBEDDED_HEADER_LINES}")

add_executable(chess_app
    src/main.c
    src/app.c
    src/game_actions.c
    src/input_handler.c
    src/input_board.c
    src/persistence.c
    src/persistence_resume.c
    src/persistence_snapshot.c
    src/ui/fonts.c
    src/ui/lobby.c
    src/ui/game.c
    src/ui/game_overlays.c
    src/ui/game_panels.c
    src/ui/game_player_panels.c
    src/ui/game_animations.c
    src/game_state.c
    src/game_state_moves.c
    src/game_notation.c
    src/lobby_state.c
    src/render/board.c
    src/network/peer.c
    src/network/session.c
    src/network/handler.c
    src/network/handler_challenges.c
    src/network/handler_connection.c
    src/network/packet_handlers.c
    src/network/packet_handshake.c
    src/network/packet_game_init.c
    src/network/packet_gameplay.c
    src/network/discovery.c
    src/network/discovery_dnssd.c
    src/network/discovery_avahi.c
    src/network/tcp.c
    src/network/tcp_messages.c
    src/network/tcp_transport.c
    ${EMBEDDED_SOURCES}
)

target_include_directories(chess_app PRIVATE include "${GENERATED_DIR}")

if(AVAHI_CLIENT_FOUND)
    target_compile_definitions(chess_app PRIVATE CHESS_APP_HAVE_AVAHI=1)
    target_include_directories(chess_app PRIVATE ${AVAHI_CLIENT_INCLUDE_DIRS})
    target_link_libraries(chess_app PRIVATE ${AVAHI_CLIENT_LIBRARIES})
endif()

# DNS-SD (Bonjour) — available on macOS as part of the system SDK, no extra link needed
if(APPLE AND NOT AVAHI_CLIENT_FOUND)
    include(CheckIncludeFile)
    check_include_file("dns_sd.h" HAVE_DNS_SD_H)
    if(HAVE_DNS_SD_H)
        target_compile_definitions(chess_app PRIVATE CHESS_APP_HAVE_DNSSD=1)
    endif()
endif()

target_link_libraries(chess_app PRIVATE SDL3::SDL3 SDL3_ttf::SDL3_ttf SDL3_image::SDL3_image Threads::Threads)

add_executable(chess_app_tests
    tests/test_suite.c
    src/game_state.c
    src/game_state_moves.c
    src/game_notation.c
    src/lobby_state.c
    src/network/tcp.c
    src/network/tcp_messages.c
    src/network/session.c
    src/network/peer.c
)

target_include_directories(chess_app_tests PRIVATE include)
target_link_libraries(chess_app_tests PRIVATE SDL3::SDL3 Threads::Threads)

add_test(NAME chess_app_tests COMMAND chess_app_tests)
